Test (ANOVA)

fire_df_2 = 
  fire_df %>% 
  select(FFMC, DMC, DC, ISI, temp, RH, wind, rain, area) %>% 
  mutate(fire = ifelse(area > 0, "fire", "no fire"))

# FFMC
fire_ffmc = 
  fire_df_2 %>%
  select(FFMC, fire)
anova_ffmc = lm(FFMC ~ factor(fire), data = fire_ffmc)
anova(anova_ffmc)
