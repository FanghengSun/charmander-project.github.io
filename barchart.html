Bar chart (month)

fire_mon = 
  fire_df %>%
  select(month, area) %>%
  filter(area > 0) %>%
  count(month) %>%
  add_row(month = 'Jan', n = 0) %>%
  add_row(month = 'Nov', n = 0) %>%
  mutate(month = stringi::stri_trans_totitle(month)) %>%
  mutate(month = factor(month, 
                        levels = c('Jan', 'Feb', 'Mar', 'Apr',
                                   'May', 'Jun', 'Jul', 'Aug',
                                   'Sep', 'Oct', 'Nov', 'Dec'),
                        ordered = TRUE))

month_df = 
  fire_df %>%
  select(month, area) %>%
  mutate(month = stringi::stri_trans_totitle(month)) %>%
  mutate(month = factor(month, 
                        levels = c('Jan', 'Feb', 'Mar', 'Apr',
                                   'May', 'Jun', 'Jul', 'Aug',
                                   'Sep', 'Oct', 'Nov', 'Dec'),
                        ordered = TRUE)) %>%
  count(month)
  

fire_m_df = merge(month_df, fire_mon, by = 'month')
fire_m_df %>%
  mutate(month = factor(month, 
                        levels = c('Jan', 'Feb', 'Mar', 'Apr',
                                   'May', 'Jun', 'Jul', 'Aug',
                                   'Sep', 'Oct', 'Nov', 'Dec'),
                        ordered = TRUE))
